"""
    regenerate_all_diagnostics()
SINGLE SOURCE OF TRUTH for all diagnostic exports.
Recomputes from saved checkpoint and writes fresh TeX macros.
"""
function regenerate_all_diagnostics()

    # Load checkpoint
    @load "cases99_full_month_checkpoint.jld2" S_daily regime_composition global_features

    # Extract current diagnostics
    D_eff_all = global_features[:, 1]
    F_W_all = global_features[:, 2]
    chi_N_all = global_features[:, 4]

    # Compute summaries
    D_eff_mean = mean(D_eff_all)
    D_eff_early = mean(D_eff_all[1:3024])        # 21 days × 144 profiles
    D_eff_late = mean(D_eff_all[3025:end])       # 10 days × 144 profiles
    D_eff_min = minimum(D_eff_all)
    D_eff_max = maximum(D_eff_all)

    S_mean = mean(S_daily)
    S_early = mean(S_daily[1:21])
    S_late = mean(S_daily[22:end])

    # Write to TeX
    open("manuscript/diagnostics_generated.tex", "w") do f
        write(f, "% AUTO-GENERATED by regenerate_all_diagnostics()\n")
        write(f, "% DO NOT EDIT MANUALLY\n")
        write(f, "% Regenerate by running: regenerate_all_diagnostics()\n\n")

        @printf(f, "\\newcommand{\\DefEffMean}{%.2f}\n", D_eff_mean)
        @printf(f, "\\newcommand{\\DefEffEarly}{%.2f}\n", D_eff_early)
        @printf(f, "\\newcommand{\\DefEffLate}{%.2f}\n", D_eff_late)
        @printf(f, "\\newcommand{\\DefEffMin}{%.2f}\n", D_eff_min)
        @printf(f, "\\newcommand{\\DefEffMax}{%.2f}\n", D_eff_max)
        @printf(f, "\\newcommand{\\SilhouetteMean}{%.3f}\n", S_mean)
        @printf(f, "\\newcommand{\\SilhouetteEarly}{%.3f}\n", S_early)
        @printf(f, "\\newcommand{\\SilhouetteLate}{%.3f}\n", S_late)
    end

    println("✓ Wrote manuscript/diagnostics_generated.tex")
    println("  D_eff: mean=$(round(D_eff_mean,digits=2)), early=$(round(D_eff_early,digits=2)), late=$(round(D_eff_late,digits=2))")
    println("  Include in manuscript preamble: \\input{diagnostics_generated.tex}")
end
